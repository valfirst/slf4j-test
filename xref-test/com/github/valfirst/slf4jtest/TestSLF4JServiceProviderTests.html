21  class TestSLF4JServiceProviderTests {

23              ServiceLoader.load(SLF4JServiceProvider.class).iterator().next();
24  
25      @BeforeEach
26      void setUp() {
27          slf4JServiceProvider.initialize();
28      }
29  
30      @Test
31      void getProviderClass() {
32          assertEquals(TestSLF4JServiceProvider.class, slf4JServiceProvider.getClass());
33      }
34  
35      @Test
36      void getLoggerFactory() {
37          assertSame(TestLoggerFactory.class, slf4JServiceProvider.getLoggerFactory().getClass());
38          assertSame(slf4JServiceProvider.getLoggerFactory(), slf4JServiceProvider.getLoggerFactory());
39      }
40  
41      @Test
42      void getLoggerFactoryReturnsCorrectlyFromSlf4JLoggerFactory() {
43          assertThat(LoggerFactory.getILoggerFactory(), is(slf4JServiceProvider.getLoggerFactory()));
44      }
45  
46      @Test
47      void getRequestedApiVersion() {
48          assertEquals("2.0.99", slf4JServiceProvider.getRequestedApiVersion());
49      }
50  
51      @Test
52      void getMarkerFactory() {
53          assertSame(BasicMarkerFactory.class, slf4JServiceProvider.getMarkerFactory().getClass());
54          assertSame(slf4JServiceProvider.getMarkerFactory(), slf4JServiceProvider.getMarkerFactory());
55      }
56  
57      @Test
58      void getMarkerFactoryReturnsCorrectlyFromSlf4JMarkerFactory() {
59          assertThat(MarkerFactory.getIMarkerFactory(), instanceOf(BasicMarkerFactory.class));
60      }
61  
62      @Test
63      void getMDCAdapter() {
64          assertSame(TestMDCAdapter.class, slf4JServiceProvider.getMDCAdapter().getClass());
65          assertSame(slf4JServiceProvider.getMDCAdapter(), slf4JServiceProvider.getMDCAdapter());
66      }
67  
68      @Test
69      void getMDCAdapterIsReturnedCorrectlyFromSlf4JMDC() {
70          assertThat(MDC.getMDCAdapter(), instanceOf(TestMDCAdapter.class));
71      }
