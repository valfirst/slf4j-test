18  @RunWith(PowerMockRunner.class)
19  public class OverridablePropertiesTests {
20  
21      private static final String PROPERTY_SOURCE_NAME = "test";
22      private static final String PROPERTY_IN_BOTH = "bothprop";
23      private static final String PROPERTY_IN_SYSTEM_PROPS = "sysprop";
24  
25      @After
26      public void resetLoggerFactory() {
27          System.getProperties().remove(PROPERTY_SOURCE_NAME + "." + PROPERTY_IN_BOTH);
28          System.getProperties().remove(PROPERTY_SOURCE_NAME + "." + PROPERTY_IN_SYSTEM_PROPS);
29      }
30  
31      @Test
32      public void propertyNotInEither() throws IOException {
33          final String defaultValue = "sensible_default";
34          OverridableProperties properties = new OverridableProperties(PROPERTY_SOURCE_NAME);
35          assertEquals(defaultValue, properties.getProperty("notpresent", defaultValue));
36      }
37  
38      @Test
39      public void propertyInFileNotInSystemProperties() throws IOException {
40          OverridableProperties properties = new OverridableProperties(PROPERTY_SOURCE_NAME);
41          assertEquals("file value", properties.getProperty("infile", "default"));
42      }
43  
44      @Test
45      public void propertyNotInFileInSystemProperties() throws IOException {
46          final String expectedValue = "system value";
47          System.setProperty(PROPERTY_SOURCE_NAME + "." + PROPERTY_IN_SYSTEM_PROPS, expectedValue);
48  
49          OverridableProperties properties = new OverridableProperties(PROPERTY_SOURCE_NAME);
50          assertEquals(expectedValue, properties.getProperty(PROPERTY_IN_SYSTEM_PROPS, "default"));
51      }
52  
53      @Test
54      public void propertyInBothFileAndSystemProperties() throws IOException {
55          final String expectedValue = "system value";
56          System.setProperty(PROPERTY_SOURCE_NAME + "." + PROPERTY_IN_BOTH, expectedValue);
57  
58          OverridableProperties properties = new OverridableProperties(PROPERTY_SOURCE_NAME);
59          assertEquals(expectedValue, properties.getProperty(PROPERTY_IN_BOTH, "default"));
60      }
61  
62      @Test
63      public void noPropertyFile() throws IOException {
64          OverridableProperties properties = new OverridableProperties("no-property-file");
65  
66          final String defaultValue = "sensible_default";
67          assertEquals(defaultValue, properties.getProperty("blah", defaultValue));
68      }
69  
70      @Test
71      @PrepareForTest(OverridableProperties.class)
72      public void ioExceptionAtPropertiesLoading() throws Exception {
73          IOException ioException = new IOException();
74          whenNew(OverridableProperties.class).withArguments(anyString()).thenThrow(ioException);
75  
76          try {
77              OverridableProperties.createUnchecked("somesource");
78              fail("UncheckedIOException was not thrown");
79          } catch (UncheckedIOException uncheckedIOException) {
80              assertThat(uncheckedIOException.getCause(), is(ioException));
81          }
82      }
83  }
