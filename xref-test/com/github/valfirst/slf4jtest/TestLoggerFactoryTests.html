251     @Test
252     void printLevelInvalidInOverridableProperties() {
253         final OverridableProperties properties = mock(OverridableProperties.class);
254         final String invalidLevelName = "nonsense";
255         when(properties.getProperty("print.level", "OFF")).thenReturn(invalidLevelName);
256 
257         final IllegalStateException illegalStateException =
258                 assertThrows(
259                         IllegalStateException.class, () -> TestLoggerFactory.createInstance(properties));
260         assertThat(
261                 illegalStateException.getMessage(),
262                 is(
263                         "Invalid level name in property print.level of file slf4jtest.properties "
264                                 + "or System property slf4jtest.print.level"));
265         assertThat(illegalStateException.getCause(), instanceOf(IllegalArgumentException.class));
266         assertThat(
267                 illegalStateException.getCause().getMessage(),
268                 is("No enum constant " + Level.class.getName() + "." + invalidLevelName));
269     }
