3   import java.io.IOException;
4   import java.io.InputStream;
5   import java.util.Properties;
6   
7   class OverridableProperties {
8       private static final Properties EMPTY_PROPERTIES = new Properties();
9       private final String propertySourceName;
10      private final Properties properties;
11  
12      OverridableProperties(final String propertySourceName) throws IOException {
13          this.propertySourceName = propertySourceName;
14          this.properties = getProperties();
15      }
16  
17      private Properties getProperties() throws IOException {
18          InputStream resourceAsStream =
19                  Thread.currentThread()
20                          .getContextClassLoader()
21                          .getResourceAsStream(propertySourceName + ".properties");
22          if (resourceAsStream != null) {
23              return loadProperties(resourceAsStream);
24          }
25          return EMPTY_PROPERTIES;
26      }
27  
28      private static Properties loadProperties(InputStream propertyResource) throws IOException {
29          try (InputStream closablePropertyResource = propertyResource) {
30              final Properties loadedProperties = new Properties();
31              loadedProperties.load(closablePropertyResource);
32              return loadedProperties;
33          }
34      }
35  
36      String getProperty(final String propertyKey, final String defaultValue) {
37          final String propertyFileProperty = properties.getProperty(propertyKey, defaultValue);
38          return System.getProperty(propertySourceName + "." + propertyKey, propertyFileProperty);
39      }
40  }
